import {
  Entity,
  Column,
  PrimaryGeneratedColumn,
  ManyToMany,
  JoinTable,
} from 'typeorm';
import { Connection } from '../../connection/connection.entity';
 
@Entity()
export class Element {
  @PrimaryGeneratedColumn()
  id: number;
 
  @Column()
  name: string;
 
  @ManyToMany(type => Connection, connection => connection.element)
  @JoinTable()
  connection: Connection[];
}
